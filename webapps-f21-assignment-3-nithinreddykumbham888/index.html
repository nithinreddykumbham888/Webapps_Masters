Dynamic programming is both a mathematical optimization method and a computer programming method.
The method was developed by Richard Bellman in the 1950s and has found applications in numerous fields, from aerospace engineering to economics. 
    int m, n;
vector dp_before(n), dp_cur(n);

long long C(int i, int j);

// compute dp_cur[l], ... dp_cur[r] (inclusive)
void compute(int l, int r, int optl, int optr) {
if (l > r)
    return;

int mid = (l + r) >> 1;
pair best = {LLONG_MAX, -1};

for (int k = optl; k <= min(mid, optr); k++) {
    best = min(best, {(k ? dp_before[k - 1] : 0) + C(k, mid), k});
}

dp_cur[mid] = best.first;
int opt = best.second;

compute(l, mid - 1, optl, opt);
compute(mid + 1, r, opt, optr);
}

int solve() {
for (int i = 0; i < n; i++)
    dp_before[i] = C(0, i);

for (int i = 1; i < m; i++) {
    compute(0, n - 1, 0, n - 1);
    dp_before = dp_cur;
}

return dp_before[n - 1];
}

    const double EPS = 1E-9;

int compute_rank(vector> A) {
int n = A.size();
int m = A[0].size();

int rank = 0;
vector row_selected(n, false);
for (int i = 0; i < m; ++i) {
    int j;
    for (j = 0; j < n; ++j) {
        if (!row_selected[j] && abs(A[j][i]) > EPS)
            break;
    }

    if (j != n) {
        ++rank;
        row_selected[j] = true;
        for (int p = i + 1; p < m; ++p)
            A[j][p] /= A[j][i];
        for (int k = 0; k < n; ++k) {
            if (k != j && abs(A[k][i]) > EPS) {
                for (int p = i + 1; p < m; ++p)
                    A[k][p] -= A[j][p] * A[k][i];
            }
        }
    }
}
return rank;
}
